Приклад 1:
#include <stdio.h>
#include <math.h>

int main() {
    double a, b, c; 
    double D;
    double x_1, x_2;

    a = 1;
    b = 3;
    c = -4;

    D = b * b - 4 * a * c;

    if (D >= 0) {
        x_1 = (-b + sqrt(D)) / (2 * a);
        x_2 = (-b - sqrt(D)) / (2 * a);

        printf("Корені рівняння: x1 = %.2f, x2 = %.2f\n", x_1, x_2);
    } else {
        printf("Рівняння не має дійсних коренів.\n");
    }

    return 0;
}

Приклад 2:
#include <stdio.h>
#include <math.h>
#include <stdlib.h>
int main() {
    double x, z;
    double result;

    // Введення значень змінних з клавіатури
    printf("Введіть значення x: ");
    scanf("%lf", &x);
    printf("Введіть значення z: ");
    scanf("%lf", &z);

    // Обчислення виразу
    result = ((z + x) / cos(x)) + sqrt(pow(fabs(sin(x) * cos(x)), 2)) + 16 * z;

    // Виведення результату
    printf("Результат обчислення виразу: %.2f\n", result);

    return 0;
}
