Приклад 8:
/* Celsius and Fahrengeit */ /* C=(5/9)*(F-32) */ 
#include<stdio.h>
#include<conio.h>
int main(){
int fahr,celsius;
int lower, upper,step; 
lower=0;
upper=300;
step=20;
fahr=lower;
printf("\n\nCelsius Fahrengeit\n");
while(fahr <= upper){
    celsius = 5*(fahr-32)/9;
    printf("%10d\t%8d\n", fahr,celsius); 
    fahr=fahr+step;
}
getch();
}

Приклад 9(програма для побудови прямокутного трикутника з "*"):
#include<stdio.h>
#include<conio.h>
int main(){
int n,i,j;
char zirochka;
printf("Введіть розмір прямокутного трикутника(висота = ширині):\n");
scanf("%d",&n);
for(i = 1; i<=n; i++){ 
    printf("\n");
    for(j=1; j<=i; j++){
        zirochka = '*';
        printf("%c",zirochka);
    }
}
}

Приклад 10(програма для побудови рівнобедреного трикутника з "*"):
#include<stdio.h>
#include<conio.h>
int main(){
int n,i,j,k; // i,y,k -- це просто лічильники
char zirochka;
printf("Введіть розмір прямокутного трикутника(висота = ширині):\n");
scanf("%d",&n);
for(i = 1; i<=n; i++){ 
    printf("\n");
    for (k = i; k < n; k++) {
        printf(" ");
    }
    for(j=1; j<=(2*i-1); j++){
        zirochka = '*';
        printf("%c",zirochka);
    }
}
}
