Приклад 1:
#include <stdio.h>

int main() {
    FILE *in; // File pointer
    int ch;   // Variable to store characters

    // Open the file for reading, check if it exists
    if ((in = fopen("sample.txt", "r")) != NULL) {
        // Read the file until the end
        while ((ch = getc(in)) != EOF) {
            // Output the character to the standard output (screen)
            putc(ch, stdout);
        }
        // Close the file
        fclose(in);
    } else {
        // Display an error message for file opening failure
        printf("File 'sample' cannot be opened\n");
    }

    return 0; // Return zero, indicating successful program completion
}

#include <stdio.h>

int main() {
    FILE *ff;
    int base;
    ff = fopen("sam.txt", "r"); // Open the file named "sam" which is identified by the pointer ff.
 
    fscanf(ff, "%d", &base); // Read an integer from the file pointed to by ff.
    fclose(ff); // Close the file.
    
    ff = fopen("data.txt", "a"); // Open the file "data" in append mode.
    fprintf(ff, "sam is %d.\n", base); // Write "sam is <value of base>." to the file pointed to by ff.
    fclose(ff); // Close the file.
    
    return 0;
}

#include <stdio.h>

int main(void) {
    FILE *pf;
    int k;

    // Open the file "proba.txt" for writing
    if ((pf = fopen("proba.txt", "w")) == NULL) {
        perror("proba.txt");
        return 1;
    }

    // Write data to the file
    for (k = 0; k <= 5; k++) {
        fprintf(pf, "%d %d\n", k, k * k * k * k);
    }

    // Close the file
    fclose(pf);

    return 0;
}

#include <stdio.h>

int main() {
    FILE *file;
    int nums[8];

    // Open the file for reading
    file = fopen("Variant_2.txt", "r");
    if (file == NULL) {
        perror("Failed to open file for reading");
        return 1;
    }

    // Read the integers from the file into the array
    for (int i = 0; i < 8; i++) {
        fscanf(file, "%d", &nums[i]);
    }

    // Close the file
    fclose(file);

    // Print the array
    printf("Array read from file:\n");
    for (int i = 0; i < 8; i++) {
        printf("%d ", nums[i]);
    }
    printf("\n");

    return 0;
}
