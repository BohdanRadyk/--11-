Приклад 8(самому написати програму):
#include <stdio.h>

int main() {
    // Задання змінних різних типів
    int i = 42;
    float f = 3.14159;
    char c = 'A';
    long l = 1234567890L;

    // Виведення змінних з різними модифікаторами специфікації перетворення

    // Виведення типу int
    printf("Integer (%%d): %d\n", i);
    printf("Integer (%%5d): %5d\n", i);  // Ширина поля 5
    printf("Integer (%%05d): %05d\n", i); // Ширина поля 5, заповнення нулями

    // Виведення типу float
    printf("Float (%%f): %f\n", f);
    printf("Float (%%.2f): %.2f\n", f);  // Два знаки після коми
    printf("Float (%%8.2f): %8.2f\n", f); // Ширина поля 8, два знаки після коми

    // Виведення типу char
    printf("Character (%%c): %c\n", c);
    printf("Character (%%3c): %3c\n", c); // Ширина поля 3

    // Виведення типу long
    printf("Long (%%ld): %ld\n", l);
    printf("Long (%%10ld): %10ld\n", l); // Ширина поля 10

    return 0;
}

#include <stdio.h>

int main() {
    // Масив з 10 символів ASCII
    char symbols[10] = {'A', 'b', '3', '!', '@', 'Z', 'z', '0', '9', '#'};

    // Виведення заголовку таблиці
    printf("Symbol  Decimal  Octal  Hexadecimal\n");
    printf("------  -------  -----  -----------\n");

    // Цикл для виведення кожного символу і його коду в різних системах числення
    for (int i = 0; i < 10; i++) {
        char symbol = symbols[i];
        printf("  %c      %3d      %03o      %02X\n", symbol, symbol, symbol, symbol);
    }

    return 0;
}

Приклад 14(самому написати програму):
#include <stdio.h>
#include <math.h>

int main() {
    double a, b, c;
    double discriminant, root1, root2, realPart, imaginaryPart;

    // Введення коефіцієнтів
    printf("Введіть коефіцієнти a, b і c для квадратного рівняння ax^2 + bx + c = 0:\n");
    printf("a = ");
    scanf("%lf", &a);
    printf("b = ");
    scanf("%lf", &b);
    printf("c = ");
    scanf("%lf", &c);

    // Обчислення дискримінанта
    discriminant = b * b - 4 * a * c;

    // Перевірка дискримінанта і обчислення коренів
    if (discriminant > 0) {
        // Два дійсних кореня
        root1 = (-b + sqrt(discriminant)) / (2 * a);
        root2 = (-b - sqrt(discriminant)) / (2 * a);
        printf("Корені рівняння є дійсними і різними:\n");
        printf("root1 = %.2lf\n", root1);
        printf("root2 = %.2lf\n", root2);
    } else if (discriminant == 0) {
        // Один дійсний корінь
        root1 = -b / (2 * a);
        printf("Корінь рівняння є дійсним і одинарним:\n");
        printf("root1 = %.2lf\n", root1);
    } else {
        // Уявні корені
        realPart = -b / (2 * a);
        imaginaryPart = sqrt(-discriminant) / (2 * a);
        printf("Корені рівняння є уявними і різними:\n");
        printf("root1 = %.2lf + %.2lfi\n", realPart, imaginaryPart);
        printf("root2 = %.2lf - %.2lfi\n", realPart, imaginaryPart);
    }

    return 0;
}

Приклад 16(самому написати програму):
#include <stdio.h>
#include <math.h>
double distance(double x1, double y1, double x2, double y2) {
    return sqrt(pow(x2 - x1, 2) + pow(y2 - y1, 2));
}
int main() {
    int k;
    printf("Введіть значення k: ");
    scanf("%d", &k);
    double x1 = 1, y1 = 1;
    double x2 = 2*k, y2 = 2*k-1;
    double x3 = -2*k, y3 = k+2;
    double a = distance(x1, y1, x2, y2);
    double b = distance(x2, y2, x3, y3);
    double c = distance(x1, y1, x3, y3);
    double p = (a + b + c) / 2;
    double S = sqrt(p * (p - a) * (p - b) * (p - c));
    double r = S / p;
    printf("Периметр: %.2f\n", 2 * p);
    printf("Площа: %.2f\n", S);
    printf("Радіус вписаного кола: %.2f\n", r);
    return 0;
}
