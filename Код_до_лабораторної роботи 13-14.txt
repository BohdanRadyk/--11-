#include<stdio.h>
#include<math.h>
double Geron (double a,double b,double c); /*оголошення функції - прототипу*/
double Geron (double a, double b, double c) { /*реалiзація  функціï*/
double p; /* локальна змінна */
p= (a+b+c)/2;
return sqrt(p*(p-a)*(p-b)*(p-c));/*обчислити вираз і повернути результат*/
}
int main() {
double u, v, w; double s;
printf ("Vvedit storonu trikutnika"); /* у консолі введіть значення 2, 3, 4 */ 
scanf("%lf %lf %lf",&u, &v,&w);
s=Geron(u,v,w);/* викликається функція Geron. До аргументів a,b,c передаються значення змінних u,v,w. Результат виклику присвоюється змінній */
printf("Ploscha 1 trikutnika %lf\n",s);
s=Geron (10.3,8.1,9.7); /* викликається функція Geron. До аргументів a,b,c передаються константи */
printf("Ploscha 2 trikutnika %lf\n",s);
s=Geron(u+10.3,v+w,w*1.7); /* викликається функція Geron. До аргументів передаються значення виразів */
printf("Ploscha 3 trikutnika %lf\n",s);
return 0;
}

#include<stdio.h>  // Підключення бібліотеки для введення/виведення
#include<math.h>   // Підключення математичної бібліотеки

// Прототип функції для обчислення дробу
double fract(int, int);

// Прототип функції для обчислення факторіалу
double factorial(int);

int main() {
    int m, n, s;  // Оголошення змінних для введення користувача
    scanf("%d %d", &m, &n);  // Зчитування двох цілих чисел від користувача
    printf("%lf\n", fract(m+1, n+1));  // Виведення результату виклику функції fract з аргументами (m+1) і (n+1)
    return 0;  // Завершення програми з кодом 0 (успішне виконання)
}

// Реалізація функції для обчислення дробу
double fract(int x, int y) {
    double t;  // Оголошення локальної змінної для збереження результату
    t = factorial(x + y) / (x * y);  // Обчислення дробу: факторіал від (x+y) поділити на добуток x і y
    return t;  // Повернення результату
}

// Реалізація функції для обчислення факторіалу
double factorial(int n) {
    int i;  // Лічильник циклу
    double p;  // Змінна для збереження результату факторіалу
    p = 1;  // Початкове значення факторіалу (факторіал від 0 або 1 дорівнює 1)
    for(i = 1; i <= n; ++i)  // Цикл для обчислення факторіалу від 1 до n
        p *= i;  // Множення p на поточне значення i
    return p;  // Повернення результату
}

#include <stdio.h>
// Функція для друкування визначеної кількості символів рядка
void printCharacters(char *str, int count) {
    for (int i = 0; i < count; i++) {
        printf("%c", str[i]);
    }
    printf("\n"); // Перехід на новий рядок після виводу вказаної кількості символів
}

int main() {
    char str[] = "Hello, World!"; // Приклад рядка для використання
    int count = 5; // Кількість символів для виводу

    printf("Вхідний рядок: %s\n", str);
    printf("Кількість символів для друкування: %d\n", count);
    printf("Виведені символи: ");
    printCharacters(str, count);

    return 0;
}
